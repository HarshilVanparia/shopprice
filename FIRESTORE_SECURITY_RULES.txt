rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is admin
    function isAdmin() {
      return request.auth.uid != null &&
             exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Allow authenticated users to read/write their own user document
    match /users/{userId} {
      allow create: if request.auth.uid != null;
      allow read: if request.auth.uid == userId || isAdmin();
      allow update: if request.auth.uid == userId || isAdmin();
      allow delete: if isAdmin();
    }

    // Items - all authenticated users can read, admins can CRUD
    match /items/{document=**} {
      allow read: if request.auth.uid != null;
      allow create, update, delete: if isAdmin();
    }

    // Categories - all authenticated users can read, admins can CRUD
    match /categories/{document=**} {
      allow read: if request.auth.uid != null;
      allow create, update, delete: if isAdmin();
    }

    // Units - all authenticated users can read, admins can CRUD
    match /units/{document=**} {
      allow read: if request.auth.uid != null;
      allow create, update, delete: if isAdmin();
    }

    // Activity logs - all authenticated users can read and create
    match /activity_logs/{document=**} {
      allow read: if request.auth.uid != null;
      allow create: if request.auth.uid != null;
    }
  }
}
